477 // Методы действий
478 int attack(int recipient_protection)
479 {
480 // Расчет множителя атаки/защиты
481 int attackDefenseDiff = m_attack - recipient_protection;
482 double multiplier = 1.0;
483
484 if (attackDefenseDiff > 0)
485 {
486 // Атака больше защиты - урон увеличивается на 5% за единицу
487 multiplier = 1.0 + (attackDefenseDiff * 0.05);
488 }
489 else if (attackDefenseDiff < 0)
490 {
491 // Защита больше атаки - урон уменьшается на 5% за единицу
492 multiplier = 1.0 / (1.0 + (abs(attackDefenseDiff) * 0.05));
493 }
494
495 // Уникальный разброс урона для каждого типа персонажа
496 double varianceRange = m_damage_variance; // 0.0 - без разброса, 1.0 - полный разброс
497 double minMultiplier = 1.0 - varianceRange;
498 double maxMultiplier = 1.0 + varianceRange;
499
500 // Генерация случайного множителя разброса
501 double randomMultiplier = minMultiplier + (static_cast<double>(rand()) / RAND_MAX) * (maxMultiplier - minMultiplier);
502
503 // Применение разброса и множителя атаки/защиты
504 double finalDamage = m_damage * randomMultiplier * multiplier;
505
506 return max(1, static_cast<int>(finalDamage)); // Минимум 1 урона
507 }

812 {
813 if (inventoryIndex < 0 || inventoryIndex >= m_inventory.size())
814 {
815 throw out_of_range("Invalid inventory index");
816 }
817
818 Item item = m_inventory[inventoryIndex];
819 if (item.getType() != ItemType::CONSUMABLE)
820 {
821 cout << "This item is not consumable.\n";
822 return;
823 }
824
825 int healthRestore = item.getStat("health_restore");
826 if (healthRestore > 0)
827 {
828 heal(healthRestore);
829 }
830
831 int staminaRestore = item.getStat("stamina_restore");
832 if (staminaRestore > 0)
833 {
834 setCurrentStamina(min(getMaxStamina(), getCurrentStamina() + staminaRestore));
835 }
836
837 m_inventory.erase(m_inventory.begin() + inventoryIndex);
838
839 cout << "Used: " << item.getName() << "\n";
840 }

857 bar += "*";

860 {
861 bar += ".";
862 }
